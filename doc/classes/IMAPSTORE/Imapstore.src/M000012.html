          def connect()
            if !@imap && @email && @password && @imap_server && @imap_port
                                if(@ssl == true)
               @imap = Net::IMAP.new(@imap_server, @imap_port, true)
                                else
               @imap = Net::IMAP.new(@imap_server, @imap_port, false)
                                end
              @imap.login(@email, @password)
            end
          end
